需求：目前项目是不支持使用mineru的官方api在线请求API的方式的，只有本地部署的mineru或者MCP服务请求和得到mineru解析的结果，需要你进行功能扩展，目前项目中已经对mineru的请求和接受都有数据
格式的要求，所以你要兼容目前的代码，使用一个额外的类去进行解耦扩展，mineru返回的结果是一个zip包，需要对其解压缩拿到里面的_content_list.json为后缀的json结果，由于后面有manule对解析结果进行切割，所以你需要看manule的组装要求去适配，
对整个mineru解析都加上日志方便追踪，以下是使用mineru的API在线请求和接收结果的代码：

    "Content-Type": "application/json",
    "Authorization": f"Bearer {token}"
}

# 自动从路径中提取文件名 (例如从 /path/to/research.pdf 提取出 research.pdf)

data = {
    "files": [
        # 使用自动提取的文件名
        {"name": file_name, "data_id": "abcd"}
    ],
    "model_version": "vlm"
}

# 使用配置的路径
file_path = [target_file]

try:
    print(f"准备上传文件: {target_file}")
    response = requests.post(url, headers=header, json=data)

    if response.status_code == 200:
        result = response.json()
        if result.get("code") == 0:  # 建议使用 .get() 防止报错
            batch_id = result["data"]["batch_id"]
            urls = result["data"]["file_urls"]
            print(f'获取预签名URL成功，batch_id: {batch_id}')

            for i in range(len(urls)):
                # 使用 'rb' 模式打开配置的路径
                with open(file_path[i], 'rb') as f:
                    print(f"正在上传到: {urls[i][:50]}...")  # 只打印URL前50字符避免刷屏
                    res_upload = requests.put(urls[i], data=f)
                    if res_upload.status_code == 200:
                        print(f"文件 {file_path[i]} 上传成功!")
                    else:
                        print(f"文件上传失败, 状态码: {res_upload.status_code}")
        else:
            print(f'API请求失败: {result.get("msg")}')
    else:
        print(f'HTTP请求失败: {response.status_code}')
except Exception as err:
    print(f"发生错误: {err}")
